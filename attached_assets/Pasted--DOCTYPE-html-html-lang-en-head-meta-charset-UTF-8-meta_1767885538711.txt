  <script>
    const products = [
      { id: 1, name: 'Espresso', size: '5lb', category: '5lb' },
      { id: 2, name: 'Double Stack', size: '5lb', category: '5lb' },
      { id: 3, name: 'Triple Stack', size: '5lb', category: '5lb' },
      { id: 4, name: 'Decaf', size: '5lb', category: '5lb' },
      { id: 5, name: 'Cold Brew', size: '5lb', category: '5lb' },
      { id: 6, name: 'Double Stack', size: '12oz', category: '12oz' },
      { id: 7, name: 'Triple Stack', size: '12oz', category: '12oz' },
      { id: 8, name: 'Decaf', size: '12oz', category: '12oz' },
      { id: 9, name: 'Espresso', size: '12oz', category: '12oz' },
    ];
    
    let orderItems = {};
    let orderHistory = [];
    let productPrices = {};
    
    async function loadDataFromServer() {
      try {
        const [pricesRes, historyRes] = await Promise.all([
          fetch('/api/prices'),
          fetch('/api/history')
        ]);
        const pricesData = await pricesRes.json();
        const historyData = await historyRes.json();
        productPrices = pricesData && typeof pricesData === 'object' ? pricesData : {};
        orderHistory = Array.isArray(historyData) ? historyData : [];
        renderPrices();
        renderHistory();
        updateSummary();
      } catch (err) {
        console.error('Error loading data:', err);
        orderHistory = [];
        productPrices = {};
      }
    }
    
    document.getElementById('vendorEmail').value = localStorage.getItem('fivestar_vendor') || '';
    document.getElementById('ccEmail').value = localStorage.getItem('fivestar_cc') || '';
    
    function renderPrices() {
      const fiveLbPrices = document.getElementById('fiveLbPrices');
      const twelveOzPrices = document.getElementById('twelveOzPrices');
      
      fiveLbPrices.innerHTML = products.filter(p => p.category === '5lb').map(p => priceRow(p)).join('');
      twelveOzPrices.innerHTML = products.filter(p => p.category === '12oz').map(p => priceRow(p)).join('');
    }
    
    function priceRow(product) {
      const price = productPrices[product.id] || '';
      return `
        <div class="price-row">
          <span class="product-name">${product.name} <span class="product-size">${product.size}</span></span>
          <input type="text" class="price-input" value="${price}" placeholder="$0.00" 
                 onchange="setPrice(${product.id}, this.value)" id="price-${product.id}">
        </div>
      `;
    }
    
    async function setPrice(id, value) {
      const price = parseFloat(value.replace(/[^0-9.]/g, '')) || 0;
      if (price === 0) {
        delete productPrices[id];
      } else {
        productPrices[id] = price;
      }
      document.getElementById('price-' + id).value = price > 0 ? price.toFixed(2) : '';
      updateSummary();
      
      try {
        await fetch('/api/prices', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ prices: { [id]: price } })
        });
      } catch (err) {
        console.error('Error saving price:', err);
      }
    }
    
    function calculateTotalCost() {
      let total = 0;
      for (const [id, qty] of Object.entries(orderItems)) {
        const price = productPrices[id] || 0;
        total += price * qty;
      }
      return total;
    }
    
    function formatCurrency(amount) {
      return '$' + amount.toFixed(2);
    }
    
    document.getElementById('vendorEmail').addEventListener('change', (e) => {
      localStorage.setItem('fivestar_vendor', e.target.value);
    });
    document.getElementById('ccEmail').addEventListener('change', (e) => {
      localStorage.setItem('fivestar_cc', e.target.value);
    });
    
    function renderProducts() {
      const fiveLb = document.getElementById('fiveLbProducts');
      const twelveOz = document.getElementById('twelveOzProducts');
      
      fiveLb.innerHTML = products.filter(p => p.category === '5lb').map(p => productRow(p)).join('');
      twelveOz.innerHTML = products.filter(p => p.category === '12oz').map(p => productRow(p)).join('');
    }
    
    function productRow(product) {
      const qty = orderItems[product.id] || 0;
      const selected = qty > 0 ? 'selected' : '';
      return `
        <div class="product-row ${selected}" id="row-${product.id}">
          <span class="product-name">${product.name} <span class="product-size">${product.size}</span></span>
          <div class="qty-controls">
            <button class="qty-btn minus" onclick="updateQty(${product.id}, -1)">-</button>
            <input type="number" class="qty-input" value="${qty || ''}" placeholder="0" 
                   onchange="setQty(${product.id}, this.value)" id="input-${product.id}">
            <button class="qty-btn plus" onclick="updateQty(${product.id}, 1)">+</button>
          </div>
        </div>
      `;
    }
    
    function updateQty(id, delta) {
      const current = orderItems[id] || 0;
      const newQty = Math.max(0, current + delta);
      setQty(id, newQty);
    }
    
    function setQty(id, value) {
      const qty = Math.max(0, parseInt(value) || 0);
      if (qty === 0) {
        delete orderItems[id];
      } else {
        orderItems[id] = qty;
      }
      renderProducts();
      updateSummary();
    }
    
    function updateSummary() {
      const items = Object.keys(orderItems).length;
      const units = Object.values(orderItems).reduce((a, b) => a + b, 0);
      const totalCost = calculateTotalCost();
      document.getElementById('itemCount').textContent = items;
      document.getElementById('totalUnits').textContent = units;
      document.getElementById('totalCost').textContent = formatCurrency(totalCost);
      document.getElementById('sendBtn').disabled = items === 0;
      document.getElementById('saveHistoryBtn').disabled = items === 0;
    }
    
    async function importLocalData() {
      const localHistory = JSON.parse(localStorage.getItem('fivestar_history') || '[]');
      const localPrices = JSON.parse(localStorage.getItem('fivestar_prices') || '{}');
      
      if (localHistory.length === 0 && Object.keys(localPrices).length === 0) {
        showNotification('No local data found on this device', 'error');
        return;
      }
      
      let imported = 0;
      
      if (Object.keys(localPrices).length > 0) {
        try {
          await fetch('/api/prices', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ prices: localPrices })
          });
          imported++;
        } catch (err) {
          console.error('Error importing prices:', err);
        }
      }
      
      for (const order of localHistory) {
        try {
          await fetch('/api/history', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              items: order.items,
              units: order.units,
              totalCost: order.totalCost || 0
            })
          });
          imported++;
        } catch (err) {
          console.error('Error importing order:', err);
        }
      }
      
      localStorage.removeItem('fivestar_history');
      localStorage.removeItem('fivestar_prices');
      
      await loadDataFromServer();
      showNotification(`Imported ${localHistory.length} orders and prices from this device`);
    }
    
    async function saveToHistoryOnly() {
      const totalUnits = Object.values(orderItems).reduce((a, b) => a + b, 0);
      
      const historyEntry = {
        items: { ...orderItems },
        units: totalUnits,
        totalCost: calculateTotalCost()
      };
      
      try {
        const historyRes = await fetch('/api/history', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(historyEntry)
        });
        const historyResult = await historyRes.json();
        historyEntry.date = historyResult.date;
        historyEntry.id = historyResult.id;
        
        if (!Array.isArray(orderHistory)) orderHistory = [];
        orderHistory.unshift(historyEntry);
        orderHistory = orderHistory.slice(0, 20);
        
        showNotification('Order saved to history');
        clearOrder();
      } catch (err) {
        showNotification('Error saving to history: ' + err.message, 'error');
      }
    }
    
    document.getElementById('saveHistoryBtn').addEventListener('click', saveToHistoryOnly);
    
    function clearOrder() {
      orderItems = {};
      document.getElementById('orderNotes').value = '';
      renderProducts();
      updateSummary();
      showNotification('Order cleared');
    }
    
    function showNotification(msg, type = 'success') {
      const el = document.getElementById('notification');
      el.textContent = msg;
      el.className = `notification ${type}`;
      setTimeout(() => el.className = 'notification hidden', 3000);
    }
    
    document.getElementById('settingsBtn').addEventListener('click', () => {
      document.getElementById('settingsPanel').classList.toggle('hidden');
      document.getElementById('historyPanel').classList.add('hidden');
      document.getElementById('settingsBtn').classList.toggle('active');
      document.getElementById('historyBtn').classList.remove('active');
    });
    
    document.getElementById('historyBtn').addEventListener('click', () => {
      document.getElementById('historyPanel').classList.toggle('hidden');
      document.getElementById('settingsPanel').classList.add('hidden');
      document.getElementById('historyBtn').classList.toggle('active');
      document.getElementById('settingsBtn').classList.remove('active');
      renderHistory();
    });
    
    document.getElementById('clearBtn').addEventListener('click', clearOrder);
    
    function renderHistory() {
      const list = document.getElementById('historyList');
      if (orderHistory.length === 0) {
        list.innerHTML = '<p style="color: #666; text-align: center; padding: 20px;">No previous orders yet</p>';
        return;
      }
      list.innerHTML = orderHistory.slice(0, 10).map(order => `
        <div class="history-item">
          <div>
            <div class="history-date">${new Date(order.date).toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' })}</div>
            <div class="history-units">${order.units} units${order.totalCost ? ' - ' + formatCurrency(order.totalCost) : ''}</div>
          </div>
          <button class="history-btn" onclick='loadOrder(${JSON.stringify(order.items)})'>Load</button>
        </div>
      `).join('');
    }
    
    function loadOrder(items) {
      orderItems = items;

    async function sendOrder() {
      const vendorEmail = document.getElementById('vendorEmail').value;
      const ccEmail = document.getElementById('ccEmail').value;
      
      if (!vendorEmail) {
        showNotification('Please set vendor email in Settings', 'error');
        document.getElementById('settingsPanel').classList.remove('hidden');
        document.getElementById('settingsBtn').classList.add('active');
        return;
      }
      
      const subject = `Weekly Coffee Order - Erwin Mills - ${new Date().toLocaleDateString()}`;
      
      let body = `Weekly Coffee Order - Erwin Mills Coffee Company\n`;
      body += `Order Date: ${new Date().toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' })}\n`;
      body += `----------------------------------------\n\n`;
      
      const fiveLb = products.filter(p => p.category === '5lb' && orderItems[p.id]);
      if (fiveLb.length) {
        body += `5LB BAGS\n--------------------\n`;
        fiveLb.forEach(p => body += `${p.name}: ${orderItems[p.id]}\n`);
        body += `\n`;
      }
      
      const twelveOz = products.filter(p => p.category === '12oz' && orderItems[p.id]);
      if (twelveOz.length) {
        body += `12OZ BAGS\n--------------------\n`;
        twelveOz.forEach(p => body += `${p.name}: ${orderItems[p.id]}\n`);
        body += `\n`;
      }
      
      const totalUnits = Object.values(orderItems).reduce((a, b) => a + b, 0);
      body += `----------------------------------------\n`;
      body += `Total Units: ${totalUnits}\n\n`;
      
      const notes = document.getElementById('orderNotes').value.trim();
      if (notes) {
        body += `NOTES\n--------------------\n${notes}\n\n`;
      }
      
      body += `Please confirm receipt of this order.\n\nThank you,\nErwin Mills Coffee Company`;
      
      document.getElementById('sendingModal').classList.remove('hidden');
      
      try {
        const response = await fetch('/send-order', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ vendorEmail, ccEmail, subject, body })
        });
        
        const result = await response.json();
        
        document.getElementById('sendingModal').classList.add('hidden');
        
        if (result.success) {
          const historyEntry = {
            items: { ...orderItems },
            units: totalUnits,
            totalCost: calculateTotalCost()
          };
          
          try {
            const historyRes = await fetch('/api/history', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify(historyEntry)
            });
            const historyResult = await historyRes.json();
            historyEntry.date = historyResult.date;
            historyEntry.id = historyResult.id;
          } catch (err) {
            historyEntry.date = new Date().toISOString();
            console.error('Error saving history:', err);
          }
          
          if (!Array.isArray(orderHistory)) orderHistory = [];
          orderHistory.unshift(historyEntry);
          orderHistory = orderHistory.slice(0, 20);
          
          document.getElementById('successModal').classList.remove('hidden');
          clearOrder();
        } else {
          showNotification(result.error || 'Failed to send order', 'error');
        }
      } catch (err) {
        document.getElementById('sendingModal').classList.add('hidden');
        showNotification('Error sending order: ' + err.message, 'error');
      }
    }

    function exportCSV() {
      if (orderHistory.length === 0) {
        showNotification('No order history to export', 'error');
        return;
      }
      
      let csv = 'Date,Units,Total Cost,Items\n';
      
      orderHistory.forEach(order => {
        const date = new Date(order.date).toLocaleDateString('en-US');
        const units = order.units;
        const cost = order.totalCost ? order.totalCost.toFixed(2) : '0.00';
        const items = Object.entries(order.items).map(([id, qty]) => {
          const product = products.find(p => p.id == id);
          return product ? `${product.name} ${product.size} x${qty}` : '';
        }).filter(Boolean).join('; ');
        
        csv += `"${date}",${units},$${cost},"${items}"\n`;
      });
      
      const blob = new Blob([csv], { type: 'text/csv' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `fivestar-orders-${new Date().toISOString().split('T')[0]}.csv`;
      a.click();
      URL.revokeObjectURL(url);
      showNotification('CSV exported successfully');
    }
    
    function exportPDF() {
      if (orderHistory.length === 0) {
        showNotification('No order history to export', 'error');
        return;
      }
      
      const { jsPDF } = window.jspdf;
      const doc = new jsPDF();
      const goldColor = [193, 154, 73];
      const pageWidth = doc.internal.pageSize.getWidth();
      
      let grandTotalUnits = 0;
      let grandTotalCost = 0;
      orderHistory.forEach(order => {
        grandTotalUnits += order.units || 0;
        grandTotalCost += order.totalCost || 0;
      });
      
      const dates = orderHistory.map(o => new Date(o.date)).sort((a, b) => a - b);
      const startDate = dates[0].toLocaleDateString('en-US', { month: 'numeric', day: 'numeric', year: 'numeric' });
      const endDate = dates[dates.length - 1].toLocaleDateString('en-US', { month: 'numeric', day: 'numeric', year: 'numeric' });
      
      doc.setFontSize(24);
      doc.setFont(undefined, 'bold');
      doc.text('Erwin Mills Coffee Co.', pageWidth / 2, 30, { align: 'center' });
      
      doc.setFontSize(16);
      doc.text('Coffee Order History', pageWidth / 2, 42, { align: 'center' });
      
      doc.setFontSize(12);
      doc.setFont(undefined, 'normal');
      doc.text(`${startDate} - ${endDate}`, pageWidth / 2, 54, { align: 'center' });
      
      doc.setFontSize(11);
      doc.text(`Total Orders: ${orderHistory.length}`, 30, 72);
      doc.text(`Total Units: ${grandTotalUnits}`, 30, 82);
      
      doc.setFont(undefined, 'bold');
      doc.text(`Total Cost: ${formatCurrency(grandTotalCost)}`, 120, 72);
      doc.setTextColor(...goldColor);
      doc.text(`Vendor: Five Star Coffee Roasters`, 120, 82);
      doc.setTextColor(0, 0, 0);
      
      let y = 100;
      
      orderHistory.forEach((order, orderIndex) => {
        if (y > 240) {
          doc.addPage();
          y = 25;
        }
        
        const orderDate = new Date(order.date).toLocaleDateString('en-US', { month: 'numeric', day: 'numeric', year: 'numeric' });
        
        doc.setFontSize(11);
        doc.setFont(undefined, 'bold');
        doc.text(`Order: ${orderDate}`, 20, y);
        y += 8;
        
        doc.setFillColor(...goldColor);
        doc.rect(20, y - 4, 170, 8, 'F');
        
        doc.setTextColor(255, 255, 255);
        doc.setFontSize(10);
        doc.text('Product', 25, y + 1);
        doc.text('Qty', 110, y + 1);
        doc.text('Unit Price', 135, y + 1);
        doc.text('Total', 170, y + 1);
        doc.setTextColor(0, 0, 0);
        doc.setFont(undefined, 'normal');
        y += 10;
        
        let orderCost = 0;
        let orderUnits = 0;
        
        Object.entries(order.items).forEach(([id, qty]) => {
          if (y > 270) {
            doc.addPage();
            y = 25;
          }
          
          const product = products.find(p => p.id == id);
          if (product) {
            const unitPrice = productPrices[id] || 0;
            const lineTotal = unitPrice * qty;
            
            doc.setFontSize(10);
            doc.text(`${product.name} (${product.size})`, 25, y);
            doc.text(String(qty), 110, y);
            doc.text(unitPrice > 0 ? formatCurrency(unitPrice) : '-', 135, y);
            doc.text(lineTotal > 0 ? formatCurrency(lineTotal) : '-', 170, y);
            
            orderUnits += qty;
            orderCost += lineTotal;
            y += 6;
          }
        });
        
        doc.setFillColor(...goldColor);
        doc.rect(20, y - 3, 170, 7, 'F');
        
        doc.setTextColor(255, 255, 255);
        doc.setFont(undefined, 'bold');
        doc.text('TOTAL', 25, y + 1);
        doc.text(String(orderUnits), 110, y + 1);
        doc.text('', 135, y + 1);
        doc.text(orderCost > 0 ? formatCurrency(orderCost) : '-', 170, y + 1);
        doc.setTextColor(0, 0, 0);
        doc.setFont(undefined, 'normal');
        
        y += 18;
      });
      
      doc.save(`coffee-orders-${new Date().toISOString().split('T')[0]}.pdf`);
      showNotification('PDF exported successfully');
    }
